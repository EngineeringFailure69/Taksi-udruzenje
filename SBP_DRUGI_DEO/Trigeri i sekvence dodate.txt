Dodati trigger-i:

create or replace TRIGGER OSOBA_AUTO_PK 
BEFORE INSERT ON OSOBA 
REFERENCING NEW AS NEW
FOR EACH ROW
BEGIN
SELECT OSOBA_ID_OSOBE_SEQ.NEXTVAL INTO :NEW.ID_OSOBE FROM DUAL;
END;
